DNS 代码块 SPF / DKIM / DMARC 一键粘贴

# MX — 路由入站邮件到本服务器
example.com.            IN  MX   10  mail.example.com.
# TXT (SPF) — 授权本服务器为该域发信
example.com.            IN  TXT  "v=spf1 ip4:203.0.113.7 ~all"
# TXT (DKIM) — 发布公钥，出站邮件已签名
haha._domainkey.example.com. IN TXT "v=DKIM1; k=rsa; p=MIGfMA0GCSqGSIb3DQEB..."
# TXT (DMARC) — 对齐策略
_dmarc.example.com.     IN  TXT  "v=DMARC1; p=reject; rua=mailto:dmarc@example.com"
